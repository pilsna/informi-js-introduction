// Issue: When the inner function executes i=5
for (var i = 1; i <= 5; i++) {
setTimeout(function() {
console.log(i); // prints 5 five times
}, 100);
}

// Use setTimeout in newer browsers and NodeJS
for (var i = 1; i <= 5; i++) {
setTimeout(function(i) {
console.log(i); // prints 1, 2, 3, 4, 5
}, 100, i);
}

// Use Lo-dash/Underscore to implement delay
for (var i = 1; i <= 5; i++) {
_.delay(function(i) {
console.log(i); // prints 1, 2, 3, 4, 5
}, 100, i);
}

// Create closure to dynamically bind variable
for (var i = 1; i <= 5; i++) {
(function(i) {
setTimeout(function() {
console.log(i); // prints 1, 2, 3, 4, 5
}, 100);
}(i));
}

Person.prototype.sayHello =
function() {
var self = this;
